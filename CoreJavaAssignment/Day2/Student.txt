package com.sonata;

public class Student {
	int stdId;
	String stdName;
	double stdMarks;
	
	Student(int id, String name, double marks){
		this.stdId= id;
		this.stdName= name;
		this.stdMarks= marks;
	}
	
	public double totalMarks(double p,double c,double m) {
		return this.stdMarks+p+c+m;
	}
	
	public void display() {
		System.out.println(stdId);
		System.out.println(stdName);
		System.out.println(stdMarks);
	}
		
	public static void main(String args[]) {
		Student s1= new Student(10,"Ananth",30);
		s1.display();
		System.out.println(s1.totalMarks(10, 20, 30));
		
		System.out.println();
		
		Student s2= new Student(20,"Siddharth",35);
		s2.display();
		System.out.println(s2.totalMarks(10, 20, 30));
	}
}
