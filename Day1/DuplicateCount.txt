public class DuplicateCount {
	public static void main(String arg[]) {
		int arr[] = {4, 1, 5, 2, 1, 5, 9, 8, 6, 5, 3, 2, 4, 7};
		int max = 0;
		int ele=0;
		for (int i = 0; i <arr.length ; i++)
		{
			int count =1;
			for (int j = i+1; j <arr.length ; j++)
			{
				if(arr[i]==arr[j])
				{
					count++;
				}
			}
			if(max<count)
			{
				max=count;
				ele = arr[i];
			}
		 }
		 System.out.println("Element repeating maximum no of times: " + ele + ", maximum count: " + max);
	}
}
